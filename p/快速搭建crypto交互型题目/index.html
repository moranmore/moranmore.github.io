def task():
    print(welcome)
    for _ in range(rounds):
        a,b = random.randint(1,1000),random.randint(1,1000)
        c = a*b
        cho = input(f"Please input your answer of {a}*{b}:")

    print(f"Here is your flag:{flag}")
